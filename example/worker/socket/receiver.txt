include("@/exception/rangeExcept")
include("socket")

func main()
    if nrows(argv) !== 1
        expectExcept("1 argument(s), got {}" format nrows(argv))
    end if

    fd = (argv row 0) slice 1
    
    if subtypeOf fd !== "int"
        typeExcept(subtypeOf fd, "int")
    end if
    
    if fd < 0
        rangeExcept(string(fd))
    end if
        
    res = undefined
    
    try
        res = recv(fd)
    catch errno
    end try

    return res
end func

return main()
