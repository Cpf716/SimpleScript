func main
    askBeforeExit = true

    sfd = server(8080, 1024)

    echo "Server listening on port 8080...\n"

    array conv = null

    conc = 0
    while 1
        while !(conv = accept(sfd))
            if conc
                echo "Disconnected.\t(0)\n"
                conc = 0
            end if
        
            continue
        end while
        
        if conc < count conv
            echo "Connected...\t({})\n" format count conv
        else if conc > count conv
            echo "Disconnected.\t({})\n" format count conv
        end if
        
        conc = count conv
        
        for cfd in conv
            send(cfd, gmt())
        end for
        
        sleep 1000
    end while
end func

main()
