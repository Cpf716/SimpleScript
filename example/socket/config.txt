//  sudo ln -s /Users/<user>/SimpleScript/example /Library/Application\ Support/SimpleScript

include("@/array/table/getPref")
include("fileSystem")

if typeOf SEP === undefined
    define SEP ","
end if

array data = readFile("/Library/Application Support/SimpleScript/example/prefs.csv", SEP)

if count data === 1
    return
end if

if typeOf FILENAME === undefined
    define FILENAME getPref(data, "filename")
end if

if typeOf PORT === undefined
    define PORT getPref(data, "port")
end if

if typeOf USER === undefined
    define USER getPref(data, "user")
end if
