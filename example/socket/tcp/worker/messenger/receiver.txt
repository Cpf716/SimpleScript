include("@/exception/rangeExcept")
include("socket")

func main()
    if nrows(argv) !== 1
        expectException("1 argument(s), got {}" format nrows(argv))
    end if

    fileDescriptor = (argv row 0) slice 1
    
    if subtypeOf fileDescriptor !== "integer"
        typeException(subtypeOf fileDescriptor, "integer")
    end if
    
    if fileDescriptor < 0
        rangeExcept(string(fileDescriptor))
    end if
        
    result = null
    
    try
        result = recv(fileDescriptor)
    catch err
    end try

    return result
end func

return main()
