include("@/array/table/stringify")

func main()
    if nrows(argv) !== 1
        expectExcept("1 argument(s), got {}" format nrows(argv))
    end if

    val = (argv row 0) slice 1

    if typeOf val === "array"
        throw "Unexpected token: {}" format stringify(((toArray val) insert 0, count val))
    end if

    return "" + val
end func

return main()
