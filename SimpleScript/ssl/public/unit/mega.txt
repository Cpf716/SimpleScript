include("@/exception/expectExcept")

func main()
    if nrows(argv) !== 1
        expectExcept("1 argument(s), got {}" format nrows(argv))
    end if
    
    num = (argv row 0) slice 1
    
    if typeOf num !== "int"
        typeExcept(typeOf num, "int")
    end if

    return num / 10^^6
end func()

return main()
