include("@/exception/expectException")

func main()
    if nrows(argv) !== 1
        expectException("1 argument(s), got {}" format nrows(argv))
    end if
    
    str = (argv row 0) slice 1

    if typeOf str !== "string"
        typeException(typeOf str, "string")
    end if

    if str === null
        throw "null"
    end if

    beg = 0
    while beg < count str && isSpace str.beg
        beg += 1
    end while

    return str substr beg
end func

return main()
