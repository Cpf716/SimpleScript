include("@/exception/expectException")

func main()
    if nrows(argv) !== 2
        expectException("2 argument(s), got {}" format nrows(argv))
    end if

    text = ((argv row 0) slice 1, (argv cell 0, 0))

    if typeOf text !== "string"
        typeException(typeOf text, "string")
    end if

    pat = ((argv row 1) slice 1, (argv cell 1, 0))

    if typeOf pat !== "string"
        typeException(typeOf pat, "string")
    end if

    if pat === null
        throw "null"
    end if

    if count text < count pat
        return 0
    end if

    beg = count text - count pat, i = 0
    while i < count pat && text.(beg + i) === pat.i
        i += 1
    end while

    return i === count pat
end func

return main()
