include("@/exception/expectException")

func main()
    if nrows(argv) !== 1
        expectException("1 argument(s), got {}" format nrows(argv))
    end if
    
    str = (argv row 0) slice 1

    if typeOf str !== "string"
        typeException(typeOf str, "string")
    end if

    if str === null
        throw "null"
    end if

    end = count str
    while end > 0 && isSpace str.(end - 1)
        end -= 1
    end while

    beg = 0
    while beg < end && isSpace str.beg
        beg += 1
    end while

    return str substr beg, end
end func

return main()
