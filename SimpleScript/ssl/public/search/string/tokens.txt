include("@/exception/expectExcept")

func main()
    if nrows(argv) !== 1
        expectExcept("1 argument(s), got {}" format nrows(argv))
    end if

    str = (argv row 0) slice 1

    if typeOf str !== "string"
        typeExcept(typeOf str, "string")
    end if

    if str === null
        throw "null"
    end if

    array arr = null, num = -1

    for end = 0; end < count str; end += 1
        while end < count str && isSpace str.end
            end += 1
        end while
        
        beg = end
        while end < count str && !isSpace str.end
            end += 1
        end while
        
        if beg != end
            arr.(num += 1) = (str substr beg, end)
        end if
    end for

    return arr
end func

return main()
