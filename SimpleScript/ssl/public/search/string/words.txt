include("@/search/string/tokens")

func main()
    if nrows(argv) !== 1
        expectException("1 argument(s), got {}" format nrows(argv))
    end if

    filename = (argv row 0) slice 1

    if typeOf filename !== "string"
        typeException(typeOf filename, "string")
    end if

    if filename === null
        throw "null"
    end if

    array arr = readFile(filename)

    arr = tokens(arr)

    if !count arr
        return
    end if

    _arr.(arr.0) = 1

    for i = 1; i < count arr; i += 1
        if _arr.(arr.i) === null
            _arr.(arr.i) = 1
        else
            _arr.(arr.i) += 1
        end if
    end for

    return _arr
end func

return main()
