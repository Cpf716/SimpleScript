include("@/exception/expectException")

func main()
    if nrows(argv) !== 1
        expectException("1 argument(s), got {}" format nrows(argv))
    end if

    array arr = (argv row 0) slice 1

    for i = 0; i < floor (count arr / 2); i += 1
        tmp = arr.i
        
        arr.i = arr.(count arr - i - 1)
        arr.(count arr - i - 1) = tmp
    end for

    return arr
end func

return main()
