include("@/math/stat/mean")

func main()
    if (count argv - 1) / first argv !== 1
        expectExcept("1 argument(s), got {}" format (count argv - 1) / first argv)
    end if

    array arr = (argv row 0) slice 1

    x = mean(arr)

    for i = 0; i < count arr; i += 1
        arr.i = (arr.i - x)^^2
    end for

    //  sample
    return sum(arr) / (count arr - 1)
end func

return main()
