include("@/sort/sort")

func main()
    if nrows(argv) !== 1
        expectException("1 argument(s), got {}" format nrows(argv))
    end if

    array arr = (argv row 0) slice 1

    arr = sort(arr)

    return arr.floor(count arr / 2)
end func

return main()
