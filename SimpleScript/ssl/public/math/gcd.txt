include("@/exception/expectException")

func main()
    if nrows(argv) !== 2
        expectException("2 argument(s), got {}" format nrows(argv))
    end if

    a = ((argv row 0) slice 1, (argv cell 0, 0))

    if subtypeOf a !== "integer"
        typeException(subtypeOf a, "integer")
    end if

    b = ((argv row 1) slice 1, (argv cell 1, 0))

    if subtypeOf b !== "integer"
        typeException(subtypeOf b, "integer")
    end if

    if a < b
        tmp = a
        a = b
        b = tmp
    end if

    r = 0
    do while r !== 0
        r = a % b
        a = b
        b = r
    end while

    return a
end func

return main()
