include("@/exception/expectException")

func main()
    if nrows(argv) !== 1
        expectException("1 argument(s), got {}" format nrows(argv))
    end if

    a = (argv row 0) slice 1

    if subtypeOf a !== "integer"
        typeException(subtypeOf a, "integer")
    end if

    func `factorial`(a, b)
        if b === 1
            return a
        end if
        
        `factorial`(a * b, b - 1)
    end func

    return `factorial`(a, a - 1)
end func

return main()
