include("@/exception/expectException")

array value = null, flag = true

func onMessage(message)
    value = message, flag = false
end func

func main()
    if nrows(argv) !== 1
        expectException("1 argument(s), got {}" format (nrows(argv)))
    end if
    
    array params = (argv row 0) slice 1
    
    if typeOf first params !== "string"
        typeException(typeOf first params, "string")
    end if
    
    if first params === null
        throw "null"
    end if
    
    worker(...params)
    
    while flag
        continue
    end while
    
    return value
end func

return main()
