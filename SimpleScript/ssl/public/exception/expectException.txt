include("@/exception/typeException")

if nrows(argv) !== 1
    throw "Expected 1 argument(s), got {}" format nrows(argv)
end if

message = (argv row 0) slice 1

if typeOf message !== "string"
    typeException(typeOf message, "string")
end if

throw "Expected " + message
