include("@/exception/expectException")

if nrows(argv) !== 1
    expectException("1 argument(s), got {}" format nrows(argv))
end if

message = (argv row 0) slice 1

if typeOf message !== "string"
    typeException(typeOf message, "string")
end if

throw "Out of range: {}" format message
