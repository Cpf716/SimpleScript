include("@/exception/expectException")

func main()
    if nrows(argv) !== 1
        expectException("1 argument(s), got {}" format nrows(argv))
    end if

    year = (argv row 0) slice 1

    if subtypeOf year !== "integer"
        typeException(subtypeOf year, "integer")
    end if

    if year < 1970
        return
    end if

    return year % 4 ? false : (year % 100 ? true : !(year % 400))
end func

return main()
