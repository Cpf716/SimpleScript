include((dir = "@/datetime/") + "date/parseDateComp")
include(dir + "weekday")
include((dir += "date/") + "day")
include(dir + "month")
include(dir + "year")

func main()
    if nrows(argv) !== 1
        expectException("Expected 1 argument(s), got {}" format nrows(argv))
    end if

    timestamp = (argv row 0) slice 1
    
    if subtypeOf timestamp !== "integer"
        typeException(subtypeOf timestamp, "integer")
    end if
    
    `month` = month(timestamp)
    `day` = day(timestamp)
    
    suppress `day`
    
    if `month` === 2
        //  March
    
        secondSunday = 1
        
        timestamp = parseDateComp("{}-03-01" format year(timestamp))
        `weekday` = weekday(timestamp)
        
        n = 0
        while true
            n += !`weekday`
            
            if n === 2
                break
            end if
            
            `weekday` = (`weekday` + 1) % 7
            secondSunday += 1
        end while
        
        return `day` >= secondSunday
    end if
    
    if `month` === 10
        //  November
        
        firstSunday = 1
        
        timestamp = parseDateComp("{}-11-01" format year(timestamp))
        `weekday` = weekday(timestamp)
        
        while `weekday`
            `weekday` = (`weekday` + 1) % 7
            firstSunday += 1
        end while
        
        return `day` < firstSunday
    end if
    
    //  February - November
    return (`month` > 1 && `month` < 10)
end func

return main()
