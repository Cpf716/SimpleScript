include("@/exception/expectException")

func main()
    if nrows(argv) !== 2
        expectException("2 argument(s), got {}" format nrows(argv))
    end if

    timestamp = ((argv row 0) slice 1, (argv cell 0, 0))

    if subtypeOf timestamp !== "integer"
        typeException(subtypeOf timestamp, "integer")
    end if

    hours = ((argv row 1) slice 1, (argv cell 1, 0))

    if subtypeOf hours !== "integer"
        typeException(subtypeOf hours, "integer")
    end if

    timestamp += hours * 3600

    return timestamp
end func

return main()
