include("@/exception/expectExcept")

func main()
    if nrows(argv) !== 2
        expectExcept("2 argument(s), got {}" format nrows(argv))
    end if

    timestamp = ((argv row 0) slice 1, (argv cell 0, 0))

    if subtypeOf timestamp !== "int"
        typeExcept(subtypeOf timestamp, "int")
    end if

    hours = ((argv row 1) slice 1, (argv cell 1, 0))

    if subtypeOf hours !== "int"
        typeExcept(subtypeOf hours, "int")
    end if

    timestamp += hours * 3600

    return timestamp
end func

return main()
