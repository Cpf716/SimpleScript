include((dir = "@/datetime/") + "constants/months")
include(dir + "date/isLeapYear")
include("@/number/digits")

func main()
    if nrows(argv) !== 2
        expectException("1 argument(s), got {}" format (nrows(argv) - 1))
    end if

    timestamp = (argv row 1) slice 1

    if subtypeOf timestamp !== "integer"
        typeException(subtypeOf timestamp, "integer")
    end if

    if timestamp < 0
        rangeExcept(string(timestamp))
    end if

    seconds = 0, secondsInYear = 0, year = 1970
    while timestamp >= seconds + (secondsInYear = (365 + isLeapYear(year)) * 86400)
        seconds += secondsInYear
        year += 1
    end while

    timestamp -= seconds
    
    days = 0
    while timestamp >= (days + 1) * 86400
        days += 1
    end while
    
    timestamp = 0
    shortYear = (digits(year) slice 2, 2)
    
    for i = 0; i < 2; i += 1
        timestamp += shortYear.i * 10^^(4 - i)
    end for
    
    timestamp += days + 1
    
    return timestamp
end func

return main()
