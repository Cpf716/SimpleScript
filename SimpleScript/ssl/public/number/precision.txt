include((dir = "@/exception/") + "expectException")
include(dir + "rangeExcept")

func main()
    if nrows(argv) !== 2
        expectException("2 argument(s), got {}" format nrows(argv))
    end if
    
    x = ((argv row 0) slice 1, (argv cell 0, 0))
    
    if typeOf x !== "number"
        typeException(typeOf x, "number")
    end if
    
    y = ((argv row 1) slice 1, (argv cell 1, 0))
    
    if subtypeOf y !== "integer"
        typeException(subtypeOf y, "integer")
    end if
    
    if y < 0
        rangeExcept(string(y))
    end if
    
    z = x - floor x
    z *= 10^^y
    w = floor z
    
    if z - w >= 1 / 2
        z += 1
    end if
    
    z = w / (10^^y)
    
    return floor x + z
end func

return main()
