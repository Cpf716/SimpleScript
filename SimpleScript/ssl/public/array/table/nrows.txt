func main()
    if (count argv - 1) / first argv !== 1
        throw "Expected 1 argument(s), got {}" format ((count argv - 1) / first argv)
    end if

    array arr = (argv row 0) slice 1

    if subtypeOf arr !== "table"
        throw "Cannot convert from {} to table" format subtypeOf arr
    end if

    return (count arr - 1) / first arr
end func

return main()
