include("@/exception/expectExcept")

func main()
    if (count argv - 1) / first argv !== 2
        expectExcept("2 argument(s), got {}" format (count argv - 1) / first argv)
    end if
    
    array arr = ((argv row 0) slice 1, (argv cell 0, 0))
    
    if subtypeOf arr !== "table"
        typeExcept(subtypeOf arr, "table")
    end if
    
    key = ((argv row 1) slice 1, (argv cell 1, 0))
    
    if typeOf key !== "string"
        typeExcept(typeOf key, "string")
    end if

    i = 0
    while i < (count arr - 1) / first arr && (arr cell i, 1) !== key
        i += 1
    end while

    if i !== (count arr - 1) / first arr
        return (val = arr row i) slice 2, first val - 1
    end if
end func

return main()
