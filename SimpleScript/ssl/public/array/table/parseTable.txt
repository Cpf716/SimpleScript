include("@/uuid")
include("fileSystem")

if typeOf SEP === null
    define SEP ","
end if

func main()
    if !nrows(argv)
        expectException("1 argument(s), got 0")
    end if
    
    if nrows(argv) >= 3
        expectException("2 argument(s), got {}" format nrows(argv))
    end if
    
    str = ((argv row 0) slice 1, (argv cell 0, 0))
    
    if typeOf str !== "string"
        typeException(typeOf str, "string")
    end if
    
    array pat = SEP
    
    if nrows(argv) === 2
        pat = ((argv row 1) slice 1, (argv cell 1, 0))
        
        if typeOf array(...pat) !== "string"
            typeException(typeOf array(...pat), "string")
        end if
    end if

    filename = "/tmp/" + uuid()

    write(filename, str)

    arr = readFile(filename, pat)
    
    remove(filename)
    
    return arr
end func

return main()
