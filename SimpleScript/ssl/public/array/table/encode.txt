include("@/exception/expectExcept")

func main()
    if (count argv - 1) / first argv !== 1
        expectExcept("1 argument(s), got {}" format (count argv - 1) / first argv)
    end if
    
    array arr = (argv row 0) slice 1
    
    if subtypeOf arr !== "table"
        typeError(subtypeOf arr, "table")
    end if
    
    arr = inverse arr
    array tmp = array(first arr)

    arr insert 1, (tmp fill (count arr - 1) / first arr)

    return inverse arr
end func

return main()
