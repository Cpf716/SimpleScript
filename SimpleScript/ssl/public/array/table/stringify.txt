include("@/array/table/encode")
include("@/uuid")
include("fileSystem")

if typeOf SEP === undefined
    define SEP "\t"
end if

//  FUNCTIONS

func main()
    if (count argv - 1) / first argv !== 1
        expectExcept("1 argument(s), got {}" format (count argv - 1) / first argv)
    end if

    array arr = (argv row 0) slice 1

    if subtypeOf arr !== "table"
        typeExcept(subtypeOf arr, "table")
    end if

    filename = "/tmp/" + uuid()
    
    arr = encode(arr)

    write(filename, arr, SEP)

    str = readFile(filename)

    remove(filename)
    
    return str
end func

return main()
